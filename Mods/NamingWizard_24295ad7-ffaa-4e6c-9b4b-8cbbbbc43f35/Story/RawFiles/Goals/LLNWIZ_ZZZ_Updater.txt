//REGION MOD_UPDATING
PROC

AND

PROC
LLNWIZ_Updater_RegisterMod((STRING)_Version)
AND
NOT DB_Mods_RegisteredMod("NamingWizard", _Version, "LaughingLeader")
THEN
LLNWIZ_Updater_ClearPreviousVersions();

LLNWIZ_QRY_Updater_ModUpdateNeeded((STRING)_NewVersion)
AND
NOT DB_Mods_RegisteredMod("NamingWizard", _NewVersion, "LaughingLeader")
THEN

IF
GameStarted(_,_)
AND
LLNWIZ_QRY_Updater_ModUpdateNeeded("0.9.0.0")
THEN

PROC
LLNWIZ_System_UpdateDatabases()
THEN

LLNWIZ_System_ClearDatabases()
THEN

//REGION UNINSTALLING
PROC
LLNWIZ_Updater_UnregisterMod()
AND
DB_Mods_RegisteredMod("NamingWizard", _Version, "LaughingLeader")
THEN
NOT DB_Mods_RegisteredMod("NamingWizard", _Version, "LaughingLeader");

PROC
LLNWIZ_Updater_UninstallMod()
THEN
LLNWIZ_DebugLog("[Updater] Uninstalling mod.");

//END_REGION
