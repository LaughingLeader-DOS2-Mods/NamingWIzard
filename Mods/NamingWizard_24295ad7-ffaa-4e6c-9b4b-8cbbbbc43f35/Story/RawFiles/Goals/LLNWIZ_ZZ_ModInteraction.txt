//Open Wizard
//StoryEvent: LLNWIZ_Events_OpenNamingWizard
//StoryEvent: LLNWIZ_Events_OpenNamingWizard_PlayerOnly
//StoryEvent: LLNWIZ_Events_NamingFinished
//Inital text var on player: LLNWIZ_InitialText
//Text after finishing: LLNWIZ_FinalText
KBSECTION
PROC
LLNWIZ_ModInteraction_SetInitialText((CHARACTERGUID)_Player)
AND
GetVarString(_Player, "LLNWIZ_InitialText", _InitialText)
THEN
LLNWIZ_Naming_SetInitialText(_InitialText, _Player);

IF
StoryEvent((CHARACTERGUID)_Player, "LLNWIZ_Events_OpenNamingWizard_PlayerOnly")
THEN
LLNWIZ_ModInteraction_SetInitialText(_Player);
LLNWIZ_Naming_OpenNamingWizard(_Player);

IF
StoryEvent((CHARACTERGUID)_Player, "LLNWIZ_Events_OpenNamingWizard")
THEN
LLNWIZ_ModInteraction_SetInitialText(_Player);
LLNWIZ_Naming_OpenNamingWizard(_Player);

IF
DB_LLNWIZ_ModBridge_RegisterWord("Common", _Word)
THEN
LLNWIZ_Naming_Words_AddModWord(_Word);
NOT DB_LLNWIZ_ModBridge_RegisterWord("Common", _Word);

IF
DB_LLNWIZ_ModBridge_RegisterWord("Player", _Word)
THEN
LLNWIZ_Naming_Words_AddPlayerWord(_Word);
NOT DB_LLNWIZ_ModBridge_RegisterWord("Player", _Word);
EXITSECTION
