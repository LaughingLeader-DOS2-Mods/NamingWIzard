THEN
LLNWIZ_Settings_InitializeFlags();

PROC
LLNWIZ_System_ClearDatabases()
THEN
SysClear("DB_LLNWIZ_ObjectFlags", 1);
SysClear("DB_LLNWIZ_GlobalFlags", 1);
SysClear("DB_LLNWIZ_UserFlags", 1);
